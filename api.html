API Reference

Full programmatic API for WebHanger.

wh.deploy(dir, name, version, options?)

const result = await wh.deploy("./components/navbar", "navbar", "1.0.0", {
  expiresInSeconds: 86400,
  token: "custom-token",
  dependencies: []
});
// { cdnUrl, token, expires }

wh.resolve(name, version?)

const comp = await wh.resolve("navbar", "1.0.0");
// { cdnUrl, token, expires, dependencies }

WebHangerFront.load()

await WebHangerFront.load(
  cdnUrl, projectId, token, expires, selector, onSignal
);

WebHangerFront.clearCache()

await WebHangerFront.clearCache();
